vue2-router 手动实现

2021-08-27 前端 JS, Vue 0 评论

vue2的router 手写

let Vue = null;
class VueRouter {
    constructor(options) {
        this.options = options
        this.current = window.location.hash.slice(1) || '/'
        Vue.util.defineReactive(this, 'current', this.current) // *
        window.addEventListener('hashchange', () => {
            this.current = window.location.hash.slice(1)

        })
    }
}
function flatCurrty(routes) {
    let n = []
    const flat = (path,children) => {
        if(children.length > 0) {
            children.map(router => {
                n.push({
                    path: path + router.path,
                    component: router.component
                })
                if(router.children) flat(path + router.path, router.children)
            })
        }
    }
    flat('', routes)
    return n
}
VueRouter.install = function (_Vue) {

    Vue = _Vue

    Vue.mixin({
        beforeCreate() {
            if (this.$options.router) {
                Vue.prototype.$routes = this.$options.router
            }
        },
    })
    Vue.component('router-view', {
        computed: {
            current() {
                return this.$routes.current
            }
        },
        render(h) {
            let nroutes = flatCurrty(this.$routes.options.routes)
            let currou = nroutes.find(i => i.path === this.current)
            if (currou.component) {
                return h(currou.component)
            } else {
                return h(null)
            }

        },
    })

    Vue.component('router-link', {
        props: {
            to: {
                type: String,
                required: true
            }
        },
        render() {
            return <a href={'#' + this.to}>{this.$slots.default[0]}</a>
        },
    })
}
export default VueRouter
